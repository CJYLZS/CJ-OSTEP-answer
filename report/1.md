{
	char * s = "/bin/ls";

	for(flag = 0;flag < MAX; ++flag) {

		if (rc < 0) {
			fprintf(stderr, "fork failed");
			exit(1);
		} else if (rc == 0) {
			switch(flag) {

				execvp(ss, sv);
				break;
			case 5:
				execvpe(ss, sv);
				break;
			default: break;
			}
		} else {
			;
		}
	}
	return 0;
}
